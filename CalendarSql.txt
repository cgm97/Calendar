DROP TABLE CALENDAR
DROP TABLE MEMBER
---- 일정 목록 테이블 ----
CREATE TABLE CALENDAR(
    CALENDARNO NUMBER(4) NOT NULL,
    LOGINID NVARCHAR2(10),
    TITLE NVARCHAR2(100) NOT NULL,
    STARTDATE NVARCHAR2(10)NOT NULL,
    ENDDATE NVARCHAR2(10)NOT NULL,
    CONTENT VARCHAR2(200),
    CONSTRAINT calendar_pk PRIMARY KEY (CALENDARNO),
    CONSTRAINT loginId_fk FOREIGN KEY (LOGINID) REFERENCES MEMBER(LOGINID) ON DELETE CASCADE
);

---- 회원 목록 테이블 ----
CREATE TABLE MEMBER(
    MEMBERNO NUMBER(4) NOT NULL,
    LOGINID NVARCHAR2(10) NOT NULL,
    LOGINPW NVARCHAR2(20)NOT NULL,
    NAME NVARCHAR2(10)Not NULL,
    EMAIL NVARCHAR2(50),
    REGDATE DATE DEFAULT SYSDATE,
    LASTLOGIN DATE DEFAULT SYSDATE,
    CONSTRAINT memberID_pk PRIMARY KEY (LOGINID),
);
