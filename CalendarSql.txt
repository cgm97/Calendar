---- 일정 목록 테이블 ----
CREATE TABLE CALENDAR(
    CALENDARNO NUMBER(4) NOT NULL PRIMARY KEY,
    TITLE NVARCHAR2(100) NOT NULL,
    STARTDATE NVARCHAR2(10)NOT NULL,
    ENDDATE NVARCHAR2(10),
    CONTENT VARCHAR2(200)
);

INSERT INTO CALENDAR (CALENDARNO,TITLE, STARTDATE , ENDDATE , CONTENT)
VALUES(CALENDARID.NEXTVAL,'제목','시작','종료','내용');

---- SEQUENCE  ( ID 값 자동 증가 함수) 생성 ----
CREATE SEQUENCE CALENDARID START WITH 1 INCREMENT BY 1 MAXVALUE 100 CYCLE NOCACHE;
---- SEQUENCE  내용 보기 ----
SELECT LAST_NUMBER FROM USER_SEQUENCES WHERE SEQUENCE_NAME = 'CALENDARID';
